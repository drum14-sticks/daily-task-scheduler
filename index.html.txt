<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>Daily Task Scheduler</title>
  <script crossorigin src="https://unpkg.com/react@18/umd/react.production.min.js"></script>
  <script crossorigin src="https://unpkg.com/react-dom@18/umd/react-dom.production.min.js"></script>
  <script src="https://unpkg.com/@babel/standalone/babel.min.js"></script>
  <script src="https://cdn.tailwindcss.com"></script>
</head>
<body>
  <div id="root"></div>
  
  <script type="text/babel">
    const { useState } = React;

    const Plus = () => <svg className="w-5 h-5" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M12 4v16m8-8H4" /></svg>;
    const Trash2 = () => <svg className="w-4 h-4" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M19 7l-.867 12.142A2 2 0 0116.138 21H7.862a2 2 0 01-1.995-1.858L5 7m5 4v6m4-6v6m1-10V4a1 1 0 00-1-1h-4a1 1 0 00-1 1v3M4 7h16" /></svg>;
    const Calendar = () => <svg className="w-6 h-6" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M8 7V3m8 4V3m-9 8h10M5 21h14a2 2 0 002-2V7a2 2 0 00-2-2H5a2 2 0 00-2 2v12a2 2 0 002 2z" /></svg>;
    const Clock = () => <svg className="w-4 h-4" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M12 8v4l3 3m6-3a9 9 0 11-18 0 9 9 0 0118 0z" /></svg>;
    const Star = () => <svg className="inline w-4 h-4 text-purple-600 ml-2" fill="currentColor" viewBox="0 0 24 24"><path d="M12 2l3.09 6.26L22 9.27l-5 4.87 1.18 6.88L12 17.77l-6.18 3.25L7 14.14 2 9.27l6.91-1.01L12 2z" /></svg>;

    function TaskScheduler() {
      const [tasks, setTasks] = useState([]);
      const [taskName, setTaskName] = useState('');
      const [taskPriority, setTaskPriority] = useState(5);
      const [taskSlots, setTaskSlots] = useState(1);
      const [focusTaskId, setFocusTaskId] = useState(null);
      const [focusSlots, setFocusSlots] = useState(2);
      const [focusStartTime, setFocusStartTime] = useState('14:00');
      const [workPeriods, setWorkPeriods] = useState([{ start: '09:00', end: '17:00' }]);
      const [lunchStart, setLunchStart] = useState('12:00');
      const [lunchEnd, setLunchEnd] = useState('12:30');
      const [breakStart, setBreakStart] = useState('15:00');
      const [breakEnd, setBreakEnd] = useState('15:30');
      const [hasBreak2, setHasBreak2] = useState(false);
      const [break2Start, setBreak2Start] = useState('17:00');
      const [break2End, setBreak2End] = useState('17:30');
      const [hasMeeting, setHasMeeting] = useState(false);
      const [meetingStart, setMeetingStart] = useState('10:00');
      const [meetingEnd, setMeetingEnd] = useState('11:00');
      const [hasMeeting2, setHasMeeting2] = useState(false);
      const [meeting2Start, setMeeting2Start] = useState('16:00');
      const [meeting2End, setMeeting2End] = useState('17:00');
      const [schedule, setSchedule] = useState([]);
      const [debugInfo, setDebugInfo] = useState('');

      const parseTime = (timeStr) => {
        const [h, m] = timeStr.split(':').map(Number);
        return { hours: h, minutes: m };
      };
      
      const formatTime = (hours, minutes) => {
        return `${String(hours).padStart(2, '0')}:${String(minutes).padStart(2, '0')}`;
      };

      const TimeInput = ({ value, onChange, label }) => {
        const { hours, minutes } = parseTime(value);
        
        return (
          <div>
            {label && <label className="text-xs text-gray-500 mb-1 block">{label}</label>}
            <div className="flex gap-1 items-center">
              <input
                type="number"
                min="0"
                max="23"
                value={hours}
                onChange={(e) => {
                  const h = Math.min(23, Math.max(0, parseInt(e.target.value) || 0));
                  onChange(formatTime(h, minutes));
                }}
                className="w-16 p-2 border rounded text-center font-medium"
                placeholder="HH"
              />
              <span className="font-bold text-gray-600">:</span>
              <input
                type="number"
                min="0"
                max="59"
                step="5"
                value={minutes}
                onChange={(e) => {
                  const m = Math.min(59, Math.max(0, parseInt(e.target.value) || 0));
                  onChange(formatTime(hours, m));
                }}
                className="w-16 p-2 border rounded text-center font-medium"
                placeholder="MM"
              />
            </div>
          </div>
        );
      };

      const addTask = () => {
        if (taskName.trim()) {
          const newTask = {
            id: Date.now(),
            name: taskName,
            priority: taskPriority,
            slots: taskSlots
          };
          setTasks([...tasks, newTask]);
          setTaskName('');
          setTaskPriority(5);
          setTaskSlots(1);
        }
      };

      const removeTask = (id) => {
        setTasks(tasks.filter(t => t.id !== id));
        if (focusTaskId === id) setFocusTaskId(null);
      };

      const timeToMinutes = (time) => {
        const [h, m] = time.split(':').map(Number);
        return h * 60 + m;
      };

      const minutesToTime = (mins) => {
        const h = Math.floor(mins / 60);
        const m = mins % 60;
        return `${String(h).padStart(2, '0')}:${String(m).padStart(2, '0')}`;
      };

      const generateSchedule = () => {
        try {
          setDebugInfo('Starting schedule generation...');
          
          if (tasks.length === 0) {
            setDebugInfo('Error: No tasks added');
            return;
          }
          if (!focusTaskId) {
            setDebugInfo('Error: No focus task selected');
            return;
          }

          const workStart = timeToMinutes(workPeriods[0].start);
          const workEnd = timeToMinutes(workPeriods[0].end);
          
          const protectedTimes = [];
          
          protectedTimes.push({
            start: timeToMinutes(lunchStart),
            end: timeToMinutes(lunchEnd),
            name: 'Lunch Time',
            type: 'lunch',
            priority: 1
          });
          
          protectedTimes.push({
            start: timeToMinutes(breakStart),
            end: timeToMinutes(breakEnd),
            name: 'Big Break 1',
            type: 'break',
            priority: 2
          });
          
          if (hasBreak2) {
            protectedTimes.push({
              start: timeToMinutes(break2Start),
              end: timeToMinutes(break2End),
              name: 'Big Break 2',
              type: 'break',
              priority: 2
            });
          }
          
          if (hasMeeting) {
            protectedTimes.push({
              start: timeToMinutes(meetingStart),
              end: timeToMinutes(meetingEnd),
              name: 'Meeting 1',
              type: 'meeting',
              priority: 0
            });
          }
          
          if (hasMeeting2) {
            protectedTimes.push({
              start: timeToMinutes(meeting2Start),
              end: timeToMinutes(meeting2End),
              name: 'Meeting 2',
              type: 'meeting',
              priority: 0
            });
          }
          
          const focusTask = tasks.find(t => t.id === focusTaskId);
          const focusStartMin = timeToMinutes(focusStartTime);
          const focusDuration = focusSlots * 60;
          
          protectedTimes.push({
            start: focusStartMin,
            end: focusStartMin + focusDuration,
            name: focusTask.name + ' (Focus)',
            type: 'focus',
            isFocus: true,
            taskId: focusTaskId,
            priority: 3
          });
          
          protectedTimes.sort((a, b) => a.start - b.start);
          
          const taskWork = new Map();
          tasks.forEach(task => {
            taskWork.set(task.id, task.slots * 50);
          });
          taskWork.set(focusTaskId, focusSlots * 50);
          
          const sortedTasks = [...tasks]
            .filter(t => t.id !== focusTaskId)
            .sort((a, b) => a.priority - b.priority);
          
          const newSchedule = [];
          
          const nonFocusProtected = protectedTimes.filter(p => !p.isFocus);
          
          for (const prot of nonFocusProtected) {
            newSchedule.push({
              startTime: minutesToTime(prot.start),
              endTime: minutesToTime(prot.end),
              activity: prot.name,
              type: prot.type,
              sortTime: prot.start
            });
          }
          
          const availableSlots = [];
          let currentMin = workStart;
          
          for (const prot of nonFocusProtected) {
            if (currentMin < prot.start) {
              availableSlots.push({ start: currentMin, end: prot.start });
            }
            currentMin = Math.max(currentMin, prot.end);
          }
          
          if (currentMin < workEnd) {
            availableSlots.push({ start: currentMin, end: workEnd });
          }
          
          const focusProt = protectedTimes.find(p => p.isFocus);
          let focusWorkRemaining = taskWork.get(focusTaskId);
          let focusSlotNum = 1;
          
          for (const slot of availableSlots) {
            if (focusWorkRemaining <= 0) break;
            
            if (slot.end <= focusProt.start) continue;
            
            let slotMin = Math.max(slot.start, focusProt.start);
            
            while (slotMin < slot.end && focusWorkRemaining > 0) {
              const timeAvailable = slot.end - slotMin;
              const workTime = Math.min(50, focusWorkRemaining, timeAvailable);
              
              if (workTime > 0) {
                newSchedule.push({
                  startTime: minutesToTime(slotMin),
                  endTime: minutesToTime(slotMin + workTime),
                  activity: `${focusTask.name} (Focus ${focusSlotNum}/${focusSlots})`,
                  type: 'focus',
                  sortTime: slotMin
                });
                
                focusWorkRemaining -= workTime;
                slotMin += workTime + 10;
                
                if (workTime >= 50) focusSlotNum++;
              } else {
                break;
              }
            }
          }
          
          taskWork.set(focusTaskId, focusWorkRemaining);
          
          let currentTaskIndex = 0;
          
          for (const slot of availableSlots) {
            let slotMin = slot.start;
            
            const focusInSlot = newSchedule.filter(s => 
              s.type === 'focus' && 
              timeToMinutes(s.startTime) >= slot.start && 
              timeToMinutes(s.startTime) < slot.end
            );
            
            let loopCount = 0;
            const MAX_LOOPS = 100;
            
            while (slotMin < slot.end && loopCount < MAX_LOOPS) {
              loopCount++;
              
              const focusHere = focusInSlot.find(f => {
                const fStart = timeToMinutes(f.startTime);
                const fEnd = timeToMinutes(f.endTime);
                return slotMin >= fStart && slotMin < fEnd + 10;
              });
              
              if (focusHere) {
                slotMin = timeToMinutes(focusHere.endTime) + 10;
                continue;
              }
              
              let taskToSchedule = null;
              let attempts = 0;
              
              if (currentTaskIndex > 0 && sortedTasks.length > 0) {
                const prevIndex = (currentTaskIndex - 1 + sortedTasks.length) % sortedTasks.length;
                const prevTask = sortedTasks[prevIndex];
                if (prevTask && taskWork.get(prevTask.id) > 0) {
                  taskToSchedule = prevTask;
                  currentTaskIndex = prevIndex;
                }
              }
              
              if (!taskToSchedule && sortedTasks.length > 0) {
                while (attempts < sortedTasks.length) {
                  const candidate = sortedTasks[currentTaskIndex % sortedTasks.length];
                  if (taskWork.get(candidate.id) > 0) {
                    taskToSchedule = candidate;
                    break;
                  }
                  currentTaskIndex++;
                  attempts++;
                }
              }
              
              if (!taskToSchedule) {
                break;
              }
              
              const workNeeded = taskWork.get(taskToSchedule.id);
              
              let nextFocusStart = slot.end;
              for (const f of focusInSlot) {
                const fStart = timeToMinutes(f.startTime);
                if (fStart > slotMin && fStart < nextFocusStart) {
                  nextFocusStart = fStart;
                }
              }
              
              const timeAvailable = Math.min(slot.end - slotMin, nextFocusStart - slotMin);
              const workTime = Math.min(50, workNeeded, timeAvailable);
              
              if (workTime >= 10) {
                newSchedule.push({
                  startTime: minutesToTime(slotMin),
                  endTime: minutesToTime(slotMin + workTime),
                  activity: taskToSchedule.name + (workTime < 50 ? ' (partial)' : ''),
                  type: 'task',
                  sortTime: slotMin
                });
                
                const newRemaining = workNeeded - workTime;
                taskWork.set(taskToSchedule.id, newRemaining);
                slotMin += workTime + 10;
                
                if (newRemaining <= 0) {
                  currentTaskIndex++;
                }
              } else {
                slotMin += 10;
              }
            }
          }
          
          newSchedule.sort((a, b) => a.sortTime - b.sortTime);
          
          setSchedule(newSchedule);
          setDebugInfo(`✅ Schedule generated! ${newSchedule.length} items`);
          
        } catch (error) {
          setDebugInfo(`❌ Error: ${error.message}`);
          console.error(error);
        }
      };

      const getActivityColor = (type) => {
        switch(type) {
          case 'focus': return 'bg-purple-100 border-purple-300';
          case 'task': return 'bg-blue-100 border-blue-300';
          case 'lunch': return 'bg-orange-100 border-orange-300';
          case 'break': return 'bg-green-100 border-green-300';
          case 'meeting': return 'bg-red-100 border-red-300';
          default: return 'bg-gray-100 border-gray-300';
        }
      };

      return (
        <div className="min-h-screen bg-gradient-to-br from-blue-50 to-indigo-100 p-6">
          <div className="max-w-7xl mx-auto">
            <h1 className="text-4xl font-bold text-center mb-8 text-indigo-900">
              📅 Daily Task Scheduler
            </h1>
            
            <div className="grid md:grid-cols-2 gap-6">
              <div className="space-y-6">
                <div className="bg-white rounded-lg shadow-lg p-6">
                  <h2 className="text-xl font-semibold mb-4 flex items-center gap-2">
                    <Plus /> Add Tasks
                  </h2>
                  <div className="space-y-3">
                    <input
                      type="text"
                      value={taskName}
                      onChange={(e) => setTaskName(e.target.value)}
                      placeholder="Task name..."
                      className="w-full p-2 border rounded"
                    />
                    <div className="flex gap-3">
                      <div className="flex-1">
                        <label className="text-sm text-gray-600">Priority (1-10)</label>
                        <input
                          type="number"
                          min="1"
                          max="10"
                          value={taskPriority}
                          onChange={(e) => setTaskPriority(Number(e.target.value))}
                          className="w-full p-2 border rounded"
                        />
                      </div>
                      <div className="flex-1">
                        <label className="text-sm text-gray-600">Slots</label>
                        <select
                          value={taskSlots}
                          onChange={(e) => setTaskSlots(Number(e.target.value))}
                          className="w-full p-2 border rounded"
                        >
                          {[1,2,3,4,5,6].map(n => <option key={n} value={n}>{n}</option>)}
                        </select>
                      </div>
                    </div>
                    <button
                      onClick={addTask}
                      className="w-full bg-blue-600 text-white py-2 rounded hover:bg-blue-700"
                    >
                      Add Task
                    </button>
                  </div>

                  <div className="mt-4 space-y-2">
                    {tasks.map(task => (
                      <div key={task.id} className="flex items-center gap-2 p-3 bg-gray-50 rounded border">
                        <input
                          type="radio"
                          name="focusTask"
                          checked={focusTaskId === task.id}
                          onChange={() => setFocusTaskId(task.id)}
                          className="w-4 h-4"
                        />
                        <div className="flex-1">
                          <span className="font-medium">{task.name}</span>
                          <span className="text-sm text-gray-600 ml-2">
                            P{task.priority} • {task.slots} slot{task.slots > 1 ? 's' : ''}
                          </span>
                          {focusTaskId === task.id && <Star />}
                        </div>
                        <button onClick={() => removeTask(task.id)} className="text-red-500">
                          <Trash2 />
                        </button>
                      </div>
                    ))}
                  </div>
                </div>

                <div className="bg-white rounded-lg shadow-lg p-6">
                  <h2 className="text-xl font-semibold mb-4">⭐ Focus Task Settings</h2>
                  <div className="space-y-3">
                    <div>
                      <label className="text-sm text-gray-600">Focus Task Slots</label>
                      <select
                        value={focusSlots}
                        onChange={(e) => setFocusSlots(Number(e.target.value))}
                        className="w-full p-2 border rounded"
                      >
                        <option value={1}>1 slot (50 min)</option>
                        <option value={2}>2 slots (100 min)</option>
                        <option value={3}>3 slots (150 min)</option>
                        <option value={4}>4 slots (200 min)</option>
                        <option value={5}>5 slots (250 min)</option>
                      </select>
                    </div>
                    <div>
                      <label className="text-sm text-gray-600 mb-2 block">Focus Start Time</label>
                      <div className="flex justify-center">
                        <TimeInput 
                          value={focusStartTime} 
                          onChange={setFocusStartTime}
                        />
                      </div>
                    </div>
                  </div>
                </div>

                <div className="bg-white rounded-lg shadow-lg p-6">
                  <h2 className="text-xl font-semibold mb-4">⏰ Time Settings</h2>
                  <div className="space-y-4">
                    <div>
                      <label className="text-sm font-medium text-gray-700 mb-2 block">Working Hours</label>
                      <div className="grid grid-cols-2 gap-4">
                        <TimeInput 
                          value={workPeriods[0].start} 
                          onChange={(val) => setWorkPeriods([{...workPeriods[0], start: val}])}
                          label="Start"
                        />
                        <TimeInput 
                          value={workPeriods[0].end} 
                          onChange={(val) => setWorkPeriods([{...workPeriods[0], end: val}])}
                          label="End"
                        />
                      </div>
                    </div>
                    
                    <div>
                      <label className="text-sm font-medium text-gray-700 mb-2 block">Lunch Time</label>
                      <div className="grid grid-cols-2 gap-4">
                        <TimeInput 
                          value={lunchStart} 
                          onChange={setLunchStart}
                          label="Start"
                        />
                        <TimeInput 
                          value={lunchEnd} 
                          onChange={setLunchEnd}
                          label="End"
                        />
                      </div>
                    </div>
                    
                    <div>
                      <label className="text-sm font-medium text-gray-700 mb-2 block">Big Break 1</label>
                      <div className="grid grid-cols-2 gap-4">
                        <TimeInput 
                          value={breakStart} 
                          onChange={setBreakStart}
                          label="Start"
                        />
                        <TimeInput 
                          value={breakEnd} 
                          onChange={setBreakEnd}
                          label="End"
                        />
                      </div>
                    </div>
                    
                    <div>
                      <label className="text-sm font-medium text-gray-700 flex items-center gap-2 mb-2">
                        <input
                          type="checkbox"
                          checked={hasBreak2}
                          onChange={(e) => setHasBreak2(e.target.checked)}
                          className="w-4 h-4"
                        />
                        Big Break 2 (optional)
                      </label>
                      {hasBreak2 && (
                        <div className="grid grid-cols-2 gap-4">
                          <TimeInput 
                            value={break2Start} 
                            onChange={setBreak2Start}
                            label="Start"
                          />
                          <TimeInput 
                            value={break2End} 
                            onChange={setBreak2End}
                            label="End"
                          />
                        </div>
                      )}
                    </div>
                    
                    <div>
                      <label className="text-sm font-medium text-gray-700 flex items-center gap-2 mb-2">
                        <input
                          type="checkbox"
                          checked={hasMeeting}
                          onChange={(e) => setHasMeeting(e.target.checked)}
                          className="w-4 h-4"
                        />
                        Meeting 1 (optional)
                      </label>
                      {hasMeeting && (
                        <div className="grid grid-cols-2 gap-4">
                          <TimeInput 
                            value={meetingStart} 
                            onChange={setMeetingStart}
                            label="Start"
                          />
                          <TimeInput 
                            value={meetingEnd} 
                            onChange={setMeetingEnd}
                            label="End"
                          />
                        </div>
                      )}
                    </div>
                    
                    <div>
                      <label className="text-sm font-medium text-gray-700 flex items-center gap-2 mb-2">
                        <input
                          type="checkbox"
                          checked={hasMeeting2}
                          onChange={(e) => setHasMeeting2(e.target.checked)}
                          className="w-4 h-4"
                        />
                        Meeting 2 (optional)
                      </label>
                      {hasMeeting2 && (
                        <div className="grid grid-cols-2 gap-4">
                          <TimeInput 
                            value={meeting2Start} 
                            onChange={setMeeting2Start}
                            label="Start"
                          />
                          <TimeInput 
                            value={meeting2End} 
                            onChange={setMeeting2End}
                            label="End"
                          />
                        </div>
                      )}
                    </div>
                  </div>
                </div>

                <button
                  onClick={generateSchedule}
                  className="w-full bg-green-600 text-white py-4 rounded-lg text-lg font-semibold hover:bg-green-700"
                >
                  🚀 Generate Schedule
                </button>

                {debugInfo && (
                  <div className="bg-yellow-50 border border-yellow-200 rounded p-4">
                    <strong>Debug:</strong> {debugInfo}
                  </div>
                )}
              </div>

              <div className="bg-white rounded-lg shadow-lg p-6">
                <h2 className="text-2xl font-semibold mb-4 flex items-center gap-2">
                  <Calendar /> Your Daily Schedule
                </h2>
                {schedule.length > 0 ? (
                  <div className="space-y-2">
                    {schedule.map((item, idx) => (
                      <div
                        key={idx}
                        className={`p-4 rounded-lg border-2 ${getActivityColor(item.type)}`}
                      >
                        <div className="flex items-center justify-between">
                          <div className="flex items-center gap-2">
                            <Clock />
                            <span className="font-semibold">
                              {item.startTime} - {item.endTime}
                            </span>
                          </div>
                          <span className="text-sm font-medium">{item.activity}</span>
                        </div>
                      </div>
                    ))}
                  </div>
                ) : (
                  <p className="text-gray-500 text-center py-8">
                    No schedule generated yet. Add tasks and click "Generate Schedule"
                  </p>
                )}
              </div>
            </div>
          </div>
        </div>
      );
    }

    ReactDOM.render(<TaskScheduler />, document.getElementById('root'));
  </script>
</body>
</html>
